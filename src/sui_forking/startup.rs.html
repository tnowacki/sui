23/// Initialize a forked network by fetching state from the remote endpoint at
24/// `forked_at_checkpoint` and starting a local Simulacrum instance.
25pub async fn initialize(
26    node: Node,
27    forked_at_checkpoint: CheckpointSequenceNumber,
28    version: &str,
29) -> Result<Context> {
30    // 1. Create DataStore — empty local cache, GraphQL wired up.
31    let data_store = DataStore::new(node.clone(), forked_at_checkpoint, version).await?;
32    let chain_identifier = data_store.get_chain_identifier();
33
34    // 2. Download and persist the startup checkpoint (summary + contents),
35    //    then read the summary back through the cache-aware getter.
36    data_store.download_and_persist_startup_checkpoint()?;
37    let checkpoint = data_store
38        .get_checkpoint_by_sequence_number(forked_at_checkpoint)?
39        .ok_or_else(|| anyhow!("checkpoint {} not found", forked_at_checkpoint))?;
40
41    // 3. Read system state — fetches object 0x5 + dynamic fields from remote via ObjectStore.
42    let system_state =
43        sui_types::sui_system_state::get_sui_system_state(&data_store).map_err(|e| {
44            anyhow!(
45                "failed to read system state at checkpoint {}: {}",
46                forked_at_checkpoint,
47                e
48            )
49        })?;
50
51    let protocol_version = system_state.protocol_version();
52
53    // 4. Build NetworkConfig with local test validators.
54    let mut rng = OsRng;
55    let config = ConfigBuilder::new_with_temp_dir()
56        .rng(&mut rng)
57        .with_chain_start_timestamp_ms(checkpoint.timestamp_ms)
58        .deterministic_committee_size(NonZeroUsize::MIN)
59        .with_protocol_version(ProtocolVersion::new(protocol_version))
60        .with_chain_override(node.chain())
61        .build();
62
63    // 5. Override validators in system state with local keys from config.
64    let system_state = override_validators(system_state, &config)?;
65
66    // 6. Build KeyStore.
67    let keystore = KeyStore::from_network_config(&config);
68
69    // 7. Create Simulacrum from custom state.
70    let simulacrum = Simulacrum::new_from_custom_state(
71        keystore,
72        checkpoint,
73        system_state,
74        &config,
75        data_store,
76        rng,
77    );
78
79    Ok(Context::new(simulacrum, chain_identifier))
80}

82/// Run the forked network
83/// context.
84pub async fn run(context: Context) -> Result<()> {
85    info!("forked network running, waiting for shutdown signal (Ctrl+C)");
86    tokio::signal::ctrl_c().await?;
87    info!("shutdown signal received, stopping forked network");
88    Ok(())
89}

91/// Replace the validator set in the system state with local validators from the NetworkConfig
92/// genesis, so the simulacrum can sign checkpoints with locally available keys.
93fn override_validators(
94    system_state: SuiSystemState,
95    config: &NetworkConfig,
96) -> Result<SuiSystemState> {
97    let genesis_validators = match config.genesis.sui_system_object() {
98        SuiSystemState::V1(inner) => inner.validators,
99        SuiSystemState::V2(inner) => inner.validators,
100        #[cfg(msim)]
101        _ => anyhow::bail!("unsupported genesis system state variant"),
102    };
103
104    match system_state {
105        SuiSystemState::V1(mut inner) => {
106            inner.validators = genesis_validators;
107            Ok(SuiSystemState::V1(inner))
108        }
109        SuiSystemState::V2(mut inner) => {
110            inner.validators = genesis_validators;
111            Ok(SuiSystemState::V2(inner))
112        }
113        #[cfg(msim)]
114        _ => anyhow::bail!("unsupported system state variant"),
115    }
