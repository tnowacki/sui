23macro_rules! block_on {
24    ($expr:expr) => {{
25        #[allow(clippy::disallowed_methods, clippy::result_large_err)]
26        {
27            if tokio::runtime::Handle::try_current().is_ok() {
28                std::thread::scope(|scope| {
29                    scope
30                        .spawn(|| {
31                            let rt = tokio::runtime::Builder::new_current_thread()
32                                .enable_all()
33                                .build()
34                                .expect("failed to build Tokio runtime");
35                            rt.block_on($expr)
36                        })
37                        .join()
38                        .expect("failed to join scoped thread running nested runtime")
39                })
40            } else {
41                let rt = tokio::runtime::Builder::new_current_thread()
42                    .enable_all()
43                    .build()
44                    .expect("failed to build Tokio runtime");
45                rt.block_on($expr)
46            }
47        }
48    }};
49}

51/// GraphQL Client for querying the GraphQL service.
52#[derive(Debug, Clone)]
53pub struct GraphQLClient {
54    client: reqwest::Client,
55    node: Node,
56    rpc: reqwest::Url,
57    version: String,
58}
59
60impl GraphQLClient {
61    /// Create a new GraphQL client
62    pub fn new(node: Node, version: &str) -> Result<Self, Error> {
63        let rpc = reqwest::Url::parse(node.gql_url())
64            .with_context(|| format!("invalid GraphQL URL '{}'", node.gql_url()))?;
65        Ok(Self {
66            client: reqwest::Client::new(),
67            node,
68            rpc,
69            version: version.to_string(),
70        })
71    }
72
73    pub(crate) async fn run_query<T, V>(
74        &self,
75        operation: &Operation<T, V>,
76    ) -> Result<GraphQlResponse<T>, Error>
77    where
78        T: serde::de::DeserializeOwned,
79        V: serde::Serialize,
80    {
81        Self::run_query_internal(&self.client, &self.rpc, &self.version, operation).await
82    }
83
84    async fn run_query_internal<T, V>(
85        client: &reqwest::Client,
86        rpc: &reqwest::Url,
87        version: &str,
88        operation: &Operation<T, V>,
89    ) -> Result<GraphQlResponse<T>, Error>
90    where
91        T: serde::de::DeserializeOwned,
92        V: serde::Serialize,
93    {
94        client
95            .post(rpc.clone())
96            .header(USER_AGENT, format!("sui-forking-v{}", version))
97            .json(operation)
98            .send()
99            .await
100            .context("Failed to send GQL query")?
101            .json::<GraphQlResponse<T>>()
102            .await
103            .context("Failed to read response in GQL query")
104    }
105
106    async fn get_verified_checkpoint_impl(
107        &self,
108        sequence_number: Option<CheckpointSequenceNumber>,
109    ) -> Result<Option<(VerifiedCheckpoint, CheckpointContents)>, Error> {
110        queries::checkpoint_query::query(sequence_number, self).await
111    }
112
113    /// Get the latest checkpoint sequence number from GraphQL RPC.
114    pub async fn get_latest_checkpoint_sequence_number(
115        &self,
116    ) -> Result<Option<CheckpointSequenceNumber>, Error> {
117        queries::latest_checkpoint_query::query(self).await
118    }
119
120    pub(crate) fn chain(&self) -> Chain {
121        match self.node {
122            Node::Mainnet => Chain::Mainnet,
123            Node::Testnet => Chain::Testnet,
124            Node::Devnet => Chain::Unknown,
125            Node::Custom(_) => Chain::Unknown,
126        }
127    }
128}

139impl ObjectRead for GraphQLClient {
140    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error> {
141        block_on!(crate::gql::queries::object_query::query(keys, self))
142    }
143}
144
145impl CheckpointRead for GraphQLClient {
146    fn get_verified_checkpoint(
147        &self,
148        sequence: Option<CheckpointSequenceNumber>,
149    ) -> Result<Option<(VerifiedCheckpoint, CheckpointContents)>, Error> {
150        Ok(block_on!(self.get_verified_checkpoint_impl(sequence))?)
151    }
152}

171    fn checkpoint_response_body(
172        certified: &sui_types::messages_checkpoint::CertifiedCheckpointSummary,
173        contents: &CheckpointContents,
174    ) -> serde_json::Value {
175        json!({
176            "data": {
177                "checkpoint": {
178                    "summaryBcs": FastCryptoBase64::from_bytes(
179                        &bcs::to_bytes(certified.data()).expect("summary should serialize"),
180                    )
181                    .encoded(),
182                    "contentBcs": FastCryptoBase64::from_bytes(
183                        &bcs::to_bytes(contents).expect("contents should serialize"),
184                    )
185                    .encoded(),
186                    "validatorSignatures": {
187                        "signature": FastCryptoBase64::from_bytes(
188                            certified.auth_sig().signature.as_ref(),
189                        )
190                        .encoded(),
191                        "signersMap": certified
192                            .auth_sig()
193                            .signers_map
194                            .iter()
195                            .map(|index| i32::try_from(index).expect("signer index fits in i32"))
196                            .collect::<Vec<_>>(),
197                    },
198                }
199            }
200        })
201    }
202
203    fn object_response_body(objects: &[Option<&Object>]) -> serde_json::Value {
204        json!({
205            "data": {
206                "multiGetObjects": objects
207                    .iter()
208                    .map(|object| {
209                        object.as_ref().map(|object| {
210                            json!({
211                                "address": object.id().to_string(),
212                                "version": object.version().value(),
213                                "objectBcs": FastCryptoBase64::from_bytes(
214                                    &bcs::to_bytes(*object).expect("object should serialize"),
215                                )
216                                .encoded(),
217                            })
218                        })
219                    })
220                    .collect::<Vec<_>>(),
221            }
222        })
223    }
224
225    fn versioned_object_at_checkpoint_response_body(object: Option<&Object>) -> serde_json::Value {
226        json!({
227            "data": {
228                "checkpoint": {
229                    "query": {
230                        "object": object.map(|object| {
231                            json!({
232                                "address": object.id().to_string(),
233                                "version": object.version().value(),
234                                "objectBcs": FastCryptoBase64::from_bytes(
235                                    &bcs::to_bytes(object).expect("object should serialize"),
236                                )
237                                .encoded(),
238                            })
239                        })
240                    }
241                }
242            }
243        })
244    }

293    async fn test_get_checkpoint_by_sequence_number() {
294        let server = MockServer::start().await;
295        let checkpoint = TestCheckpointBuilder::new(11).build_checkpoint();
296
297        Mock::given(method("POST"))
298            .and(path("/"))
299            .and(body_partial_json(json!({

304            .respond_with(
305                ResponseTemplate::new(200).set_body_json(checkpoint_response_body(
306                    &checkpoint.summary,
307                    &checkpoint.contents,
308                )),
309            )
310            .mount(&server)
311            .await;
312
313        let store = mock_store(&server);
314        let (verified, contents) = store
315            .get_verified_checkpoint_impl(Some(11))
316            .await
317            .expect("checkpoint query should succeed")
318            .expect("checkpoint should be present");
319
320        assert_eq!(verified.data(), checkpoint.summary.data());
321        assert_eq!(
322            verified.auth_sig().epoch,
323            checkpoint.summary.auth_sig().epoch
324        );
325        assert_eq!(
326            verified.auth_sig().signature.as_ref(),
327            checkpoint.summary.auth_sig().signature.as_ref()
328        );
329        assert_eq!(
330            verified.auth_sig().signers_map,
331            checkpoint.summary.auth_sig().signers_map
332        );
333        assert_eq!(contents.digest(), checkpoint.contents.digest());
334    }

351                            "version": versioned_object.version().value(),
352                        },
353                        {
354                            "address": root_version_object.id().to_string(),

356                        },

358                            "address": missing_object_id.to_string(),
359                            "atCheckpoint": 29,
360                        },

362                }
363            })))
364            .respond_with(
365                ResponseTemplate::new(200).set_body_json(object_response_body(&[

367                    Some(&root_version_object),

369                ])),

372            .await;
