34/// Transaction data with effects and checkpoint.
35#[derive(Clone, Debug)]
36pub(crate) struct TransactionInfo {
37    pub(crate) data: TransactionData,
38    pub(crate) effects: TransactionEffects,
39    pub(crate) checkpoint: u64,
40}
41
42/// `TransactionRead` trait is used to retrieve transaction data for a given digest.
43pub(crate) trait TransactionRead {
44    /// Given a transaction digest, return transaction info including data, effects,
45    /// and the checkpoint that transaction was executed in.
46    /// Returns `None` if the transaction is not found.
47    fn transaction_data_and_effects(
48        &self,
49        tx_digest: &str,
50    ) -> Result<Option<TransactionInfo>, Error>;
51}

53/// Query for an object.
54/// Specifies an `ObjectID` and the "rule" to retrieve it.
55#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
56pub(crate) struct ObjectKey {
57    pub(crate) object_id: ObjectID,
58    pub(crate) version_query: VersionQuery,
59}
60
61/// Query options for an object.
62/// `Version` request an object at a specific version, or latest if no version is provided
63/// `RootVersion` request an object at a given version at most (<=)
64/// `AtCheckpoint` request an object at a given checkpoint. Useful for unknown `Version`.
65/// `VersionAtCheckpoint` requests an exact version, but only if it existed by the given checkpoint.
66#[derive(Clone, Debug, PartialEq, Eq, PartialOrd, Ord)]
67pub(crate) enum VersionQuery {
68    Version(u64),
69    RootVersion(u64),
70    AtCheckpoint(u64),
71    VersionAtCheckpoint { version: u64, checkpoint: u64 },
72}
73
74/// The `ObjectRead` trait is used to retrieve objects by their keys, with different query options.
75pub(crate) trait ObjectRead {
76    /// Retrieve objects by their keys, with different query options.
77    ///
78    /// If the object is not found, the element in the vector is `None`.
79    /// Otherwise each tuple contains:
80    /// - `Object`: The object data
81    /// - `u64`: The actual version of the object
82    fn get_objects(&self, keys: &[ObjectKey]) -> Result<Vec<Option<(Object, u64)>>, Error>;
83}
84
85/// Checkpoint read data.
86pub(crate) trait CheckpointRead {
87    /// Return the verified checkpoint summary together with its decoded
88    /// contents. If `sequence` is `None`, return the latest checkpoint.
89    fn get_verified_checkpoint(
90        &self,
91        sequence: Option<CheckpointSequenceNumber>,
92    ) -> Result<Option<(VerifiedCheckpoint, CheckpointContents)>, Error>;
93}
