70impl QueryOptions {
71    pub fn from_proto(
72        request: Option<&ProtoQueryOptions>,
73        default_limit_items: u32,
74        max_limit_items: u32,
75        query_type: QueryType,
76    ) -> Result<Self, RpcError> {
77        let limit_items = request
78            .and_then(|options| options.limit_items)
79            .unwrap_or(default_limit_items)
80            .clamp(1, max_limit_items) as usize;
81
82        let ordering = match request.map(|options| options.ordering) {
83            None | Some(ORDERING_ASCENDING) => Ordering::Ascending,
84            Some(ORDERING_DESCENDING) => Ordering::Descending,
85            Some(_) => {
86                return Err(FieldViolation::new("options.ordering")
87                    .with_description("invalid ordering")
88                    .with_reason(ErrorReason::FieldInvalid)
89                    .into());
90            }
91        };
92
93        let after = parse_cursor(
94            "options.after",
95            request.and_then(|options| options.after.as_ref()),
96            query_type,
97        )?;
98        let before = parse_cursor(
99            "options.before",
100            request.and_then(|options| options.before.as_ref()),
101            query_type,
102        )?;
103
104        Ok(Self {
105            query_type,
106            limit_items,
107            ordering,
108            after,
109            before,
110        })
111    }
112
113    pub fn scan_direction(&self) -> ScanDirection {
114        match self.ordering {
115            Ordering::Ascending => ScanDirection::Ascending,
116            Ordering::Descending => ScanDirection::Descending,
117        }
118    }
119
120    pub fn is_ascending(&self) -> bool {
121        matches!(self.ordering, Ordering::Ascending)
122    }
123
124    /// Whether the request explicitly positioned the low end of the scan via an
125    /// `after` cursor. `apply_cursor_bounds` only ever raises `range.start` from
126    /// `after` (in both orderings); `before` bounds the high end. Together with an
127    /// explicit `start_checkpoint`, this lets the pruning-floor check distinguish
128    /// "resume/start from here" (error if below the floor — the data is gone) from
129    /// an open-ended low end (clamp up to the floor).
130    pub fn has_after_cursor(&self) -> bool {
131        self.after.is_some()
132    }
133
134    pub fn apply_cursor_bounds(&self, resolved: ResolvedRange) -> ResolvedRange {
135        if resolved.is_empty() {
136            return resolved;
137        }
138
139        let mut start = resolved.range.start;
140        let mut end = resolved.range.end;
141        let mut end_checkpoint = resolved.end_checkpoint;
142        let mut end_position = resolved.end_position;
143        let mut end_reason = resolved.end_reason;
144        let mut cursor_terminal = None;
145
146        if let Some(cursor) = &self.after {
147            let Some(after) = cursor.after_position_start() else {
148                return ResolvedRange::empty_at(
149                    cursor.checkpoint,
150                    cursor.position,
151                    QueryEndReason::CursorBound,
152                );
153            };
154            if after >= start {
155                start = after;
156                if matches!(self.ordering, Ordering::Descending) || after >= end {
157                    cursor_terminal = Some((cursor.checkpoint, after));
158                }
159                if matches!(self.ordering, Ordering::Descending) {
160                    end_checkpoint = cursor.checkpoint;
161                    end_position = after;
162                    end_reason = QueryEndReason::CursorBound;
163                }
164            }
165        }
166
167        if let Some(cursor) = &self.before
168            && cursor.position <= end
169        {
170            end = cursor.position;
171            if matches!(self.ordering, Ordering::Ascending) || cursor.position <= start {
172                cursor_terminal = Some((cursor.checkpoint, cursor.position));
173            }
174            if matches!(self.ordering, Ordering::Ascending) {
175                end_checkpoint = cursor.checkpoint;
176                end_position = cursor.position;
177                end_reason = QueryEndReason::CursorBound;
178            }
179        }
180
181        if start >= end {
182            if let Some((checkpoint, position)) = cursor_terminal {
183                end_checkpoint = checkpoint;
184                end_position = position;
185            }
186            if self.after.is_some() || self.before.is_some() {
187                end_reason = QueryEndReason::CursorBound;
188            }
189            ResolvedRange::empty_at(end_checkpoint, end_position, end_reason)
190        } else {
191            ResolvedRange {
192                range: start..end,
193                end_checkpoint,
194                end_position,
195                end_reason,
196            }
197        }
198    }
199
200    pub fn cursor_for_item(&self, checkpoint: u64, position: u64) -> Bytes {
201        self.encode_cursor(CursorKind::Item, checkpoint, position)
202    }
203
204    pub fn cursor_for_boundary(&self, checkpoint: u64, position: u64) -> Bytes {
205        self.encode_cursor(CursorKind::Boundary, checkpoint, position)
206    }

208    fn encode_cursor(&self, kind: CursorKind, checkpoint: u64, position: u64) -> Bytes {
209        encode_cursor(CursorToken {
210            query_type: self.query_type,
211            kind,
212            checkpoint,
213            position,
214        })
215    }

218impl ResolvedCheckpointRange {
219    pub fn empty_at(checkpoint: u64, reason: QueryEndReason) -> Self {
220        Self {
221            range: checkpoint..checkpoint,
222            end_reason: reason,
223        }
224    }

226    pub fn is_empty(&self) -> bool {
227        self.range.is_empty()
228    }
229
230    pub fn terminal_checkpoint(&self, ordering: Ordering) -> u64 {
231        match ordering {
232            Ordering::Ascending => self.range.end,
233            Ordering::Descending => self.range.start,
234        }
235    }
236
237    pub fn with_range(self, range: Range<u64>, ordering: Ordering) -> ResolvedRange {
238        let end_position = match ordering {
239            Ordering::Ascending => range.end,
240            Ordering::Descending => range.start,
241        };
242        ResolvedRange {
243            range,
244            end_checkpoint: self.terminal_checkpoint(ordering),
245            end_position,
246            end_reason: self.end_reason,
247        }
248    }
249}

270impl CheckpointRange {
271    pub fn from_request(
272        start_checkpoint: Option<u64>,
273        end_checkpoint: Option<u64>,
274        checkpoint_hi_exclusive: u64,
275    ) -> Result<Self, RpcError> {
276        let start = start_checkpoint.unwrap_or(0);
277        if let Some(end) = end_checkpoint
278            && end < start
279        {
280            return Err(FieldViolation::new("end_checkpoint")
281                .with_description(
282                    "end_checkpoint must be greater than or equal to start_checkpoint",
283                )
284                .with_reason(ErrorReason::FieldInvalid)
285                .into());
286        }
287
288        let requested_end = end_checkpoint.unwrap_or(checkpoint_hi_exclusive);
289        let high_reason = if end_checkpoint.is_none() || requested_end > checkpoint_hi_exclusive {
290            QueryEndReason::LedgerTip
291        } else {
292            QueryEndReason::CheckpointBound
293        };
294        let end = requested_end.min(checkpoint_hi_exclusive);
295
296        Ok(Self {
297            start,
298            end,
299            high_reason,
300            indexed_tip: checkpoint_hi_exclusive,
301        })
302    }
303
304    pub fn resolve(self, options: &QueryOptions) -> ResolvedCheckpointRange {
305        let mut start = self.start;
306        let mut end = self.end;
307        let mut low_reason = QueryEndReason::CheckpointBound;
308        let mut high_reason = self.high_reason;
309        let mut cursor_bound = false;
310
311        if let Some(cursor) = &options.after
312            && cursor.checkpoint >= start
313        {
314            start = cursor.checkpoint;
315            cursor_bound = true;
316            if matches!(options.ordering, Ordering::Descending) {
317                low_reason = QueryEndReason::CursorBound;
318            }
319        }
320
321        if let Some(cursor) = &options.before
322            && let Some(upper) = cursor.before_checkpoint_end()
323            && upper <= end
324        {
325            end = upper;
326            cursor_bound = true;
327            if matches!(options.ordering, Ordering::Ascending) {
328                high_reason = QueryEndReason::CursorBound;
329            }
330        }
331
332        if start >= self.indexed_tip {
333            return ResolvedCheckpointRange::empty_at(self.indexed_tip, QueryEndReason::LedgerTip);
334        }
335
336        if start >= end {
337            let reason = if cursor_bound {
338                QueryEndReason::CursorBound
339            } else {
340                match options.ordering {
341                    Ordering::Ascending => high_reason,
342                    Ordering::Descending => low_reason,
343                }
344            };
345            let checkpoint = match options.ordering {
346                Ordering::Ascending => end,
347                Ordering::Descending => start,
348            };
349            return ResolvedCheckpointRange::empty_at(checkpoint, reason);
350        }
351
352        let end_reason = match options.ordering {
353            Ordering::Ascending => high_reason,
354            Ordering::Descending => low_reason,
355        };
356        ResolvedCheckpointRange {
357            range: start..end,
358            end_reason,
359        }
360    }
361}

363#[derive(Clone, Debug, Eq, PartialEq, serde::Serialize, serde::Deserialize)]
364struct CursorToken {
365    query_type: QueryType,
366    kind: CursorKind,
367    checkpoint: u64,
368    position: u64,
369}
370
371impl CursorToken {
372    fn validate(&self, query_type: QueryType) -> bool {
373        self.query_type == query_type
374    }
375
376    fn after_position_start(&self) -> Option<u64> {
377        match self.kind {
378            CursorKind::Item => self.position.checked_add(1),
379            CursorKind::Boundary => Some(self.position),
380        }
381    }
382
383    fn before_checkpoint_end(&self) -> Option<u64> {
384        match self.kind {
385            CursorKind::Item => self.checkpoint.checked_add(1),
386            CursorKind::Boundary => Some(self.checkpoint),
387        }
388    }
389}
390
391fn parse_cursor(
392    field: &'static str,
393    cursor: Option<&Bytes>,
394    query_type: QueryType,
395) -> Result<Option<CursorToken>, RpcError> {
396    cursor
397        .map(|cursor| decode_cursor(field, cursor))
398        .transpose()?
399        .map(|token| {
400            if token.validate(query_type) {
401                Ok(token)
402            } else {
403                Err(invalid_cursor(field, "invalid cursor"))
404            }
405        })
406        .transpose()
407}
408
409fn decode_cursor(field: &'static str, cursor: &[u8]) -> Result<CursorToken, RpcError> {
410    bcs::from_bytes(cursor).map_err(|_| invalid_cursor(field, "invalid cursor"))
411}
412
413fn encode_cursor(cursor: CursorToken) -> Bytes {
414    bcs::to_bytes(&cursor).unwrap().into()
415}
416
417fn invalid_cursor(field: &'static str, description: impl Into<String>) -> RpcError {
418    FieldViolation::new(field)
419        .with_description(description)
420        .with_reason(ErrorReason::FieldInvalid)
421        .into()
422}
