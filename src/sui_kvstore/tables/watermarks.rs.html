4//! Watermarks table: stores per-pipeline watermarks indexed by pipeline name.
5//!
6//! A row in the `v1` schema has one cell per [`WatermarkV1`] field (u64 big-endian), plus a
7//! schema-version tag cell and an optional BCS-encoded [`WatermarkV0`] cell:
8//! - `ehi` / `chi` / `th` / `tmhi` / `rl` / `ph` / `ptm`: per-field u64 BE cells. `chi`
9//!   (checkpoint_hi_inclusive) is absent when the committer has not observed a checkpoint
10//!   yet (`Option<u64>::None`).
11//! - `v`: u64 BE schema version. Rows in the current schema carry `v = SCHEMA_V1`;
12//!   `create_pipeline_watermark_if_absent` keys off this cell to detect row existence,
13//!   and a future format migration can bump it to branch read behavior.
14//! - `w` (v0): BCS-encoded [`WatermarkV0`], kept in sync alongside the per-field cells so
15//!   existing BCS consumers keep working.
16
17use anyhow::Context;
18use anyhow::Result;
19use anyhow::bail;
20use bytes::Bytes;
21
22use crate::WatermarkV0;
23use crate::WatermarkV1;

25pub mod col {
26    /// BCS-encoded v0 watermark column.
27    pub const WATERMARK_V0: &str = "w";
28    /// Schema version tag for the `v1` per-field cell layout.
29    pub const SCHEMA_VERSION: &str = "v";
30    /// v1 per-field cells (all u64 big-endian).
31    pub const EPOCH_HI: &str = "ehi";
32    pub const CHECKPOINT_HI: &str = "chi";
33    pub const TX_HI: &str = "th";
34    pub const TIMESTAMP_MS_HI: &str = "tmhi";
35    pub const READER_LO: &str = "rl";
36    pub const PRUNER_HI: &str = "ph";
37    pub const PRUNER_TIMESTAMP_MS: &str = "ptm";
38    /// Per-pipeline chain id (raw 32 bytes). Independent of the `v1` schema cells —
39    /// written by `accepts_chain_id` and ignored by watermark decoding.
40    pub const CHAIN_ID: &str = "cid";
41}
42
43/// Current schema version written into the `v` cell.
44pub const SCHEMA_V1: u64 = 1;
45
46pub const NAME: &str = "watermark_alt";

48pub fn encode_key(pipeline: &str) -> Vec<u8> {
49    pipeline.as_bytes().to_vec()
50}

52/// Single `(w, BCS)` cell. Used by tests that seed v0-format data.
53pub fn encode_v0(v0: &WatermarkV0) -> Result<(&'static str, Bytes)> {
54    Ok((col::WATERMARK_V0, Bytes::from(bcs::to_bytes(v0)?)))
55}

57fn decode_u64_be(val: &Bytes, col_name: &str) -> Result<u64> {
58    if val.len() != 8 {
59        bail!(
60            "`{}` column has unexpected length {} (expected 8)",
61            col_name,
62            val.len()
63        );
64    }
65    let mut buf = [0u8; 8];
66    buf.copy_from_slice(val);
67    Ok(u64::from_be_bytes(buf))
68}

70/// Strict read of the v1 cells. Returns `Ok(None)` if the schema-version `v` cell is absent
71/// (v0-only row or row does not exist). Bails on unknown schema versions, and on missing
72/// required field cells when `v` is present (which would indicate data corruption since all
73/// field cells are written atomically with `v`). `chi` is the only optional field.
74pub fn decode_v1(row: &[(Bytes, Bytes)]) -> Result<Option<WatermarkV1>> {
75    let mut schema_version: Option<&Bytes> = None;
76    let mut epoch_hi: Option<&Bytes> = None;
77    let mut checkpoint_hi: Option<&Bytes> = None;
78    let mut tx_hi: Option<&Bytes> = None;
79    let mut timestamp_ms_hi: Option<&Bytes> = None;
80    let mut reader_lo: Option<&Bytes> = None;
81    let mut pruner_hi: Option<&Bytes> = None;
82    let mut pruner_timestamp_ms: Option<&Bytes> = None;
83    for (col, val) in row {
84        match col.as_ref() {
85            b if b == col::SCHEMA_VERSION.as_bytes() => schema_version = Some(val),
86            b if b == col::EPOCH_HI.as_bytes() => epoch_hi = Some(val),
87            b if b == col::CHECKPOINT_HI.as_bytes() => checkpoint_hi = Some(val),
88            b if b == col::TX_HI.as_bytes() => tx_hi = Some(val),
89            b if b == col::TIMESTAMP_MS_HI.as_bytes() => timestamp_ms_hi = Some(val),
90            b if b == col::READER_LO.as_bytes() => reader_lo = Some(val),
91            b if b == col::PRUNER_HI.as_bytes() => pruner_hi = Some(val),
92            b if b == col::PRUNER_TIMESTAMP_MS.as_bytes() => pruner_timestamp_ms = Some(val),
93            _ => {}
94        }
95    }
96    let Some(v) = schema_version else {
97        return Ok(None);
98    };
99    let schema = decode_u64_be(v, col::SCHEMA_VERSION)?;
100    if schema != SCHEMA_V1 {
101        bail!("unknown watermark schema version {}", schema);
102    }
103    let missing = |name: &str| anyhow::anyhow!("`{}` column missing from v1 watermark row", name);
104    let decode_required = |cell: Option<&Bytes>, name: &'static str| -> Result<u64> {
105        decode_u64_be(cell.ok_or_else(|| missing(name))?, name)
106    };
107    Ok(Some(WatermarkV1 {
108        epoch_hi_inclusive: decode_required(epoch_hi, col::EPOCH_HI)?,
109        checkpoint_hi_inclusive: checkpoint_hi
110            .map(|v| decode_u64_be(v, col::CHECKPOINT_HI))
111            .transpose()?,
112        tx_hi: decode_required(tx_hi, col::TX_HI)?,
113        timestamp_ms_hi_inclusive: decode_required(timestamp_ms_hi, col::TIMESTAMP_MS_HI)?,
114        reader_lo: decode_required(reader_lo, col::READER_LO)?,
115        pruner_hi: decode_required(pruner_hi, col::PRUNER_HI)?,
116        pruner_timestamp_ms: decode_required(pruner_timestamp_ms, col::PRUNER_TIMESTAMP_MS)?,
117    }))
118}

120/// Reads only the v0 `w` cell. Returns `Ok(None)` if absent. Used **only** by
121/// `init_watermark` to bootstrap the v1 format from a v0-only row.
122pub fn decode_v0(row: &[(Bytes, Bytes)]) -> Result<Option<WatermarkV0>> {
123    for (col, val) in row {
124        if col.as_ref() == col::WATERMARK_V0.as_bytes() {
125            return Ok(Some(bcs::from_bytes(val).context(
126                "failed to deserialize BCS v0 watermark from `w` column",
127            )?));
128        }
129    }
130    Ok(None)
131}
