4//! BigTable Store implementation for sui-indexer-alt-framework.
5//!
6//! This implements the `Store`, `Connection`, and `ConcurrentConnection` traits to allow the
7//! new framework to use BigTable for watermark storage. Per-pipeline watermarks are stored in
8//! the `watermark_alt` table as:
9//! - `w` (BCS v0 `WatermarkV0`) — kept in sync for backward compatibility.
10//! - `v` (schema version, currently `1`) — marks the row as using the new schema.
11//! - `ehi` / `chi` / `th` / `tmhi` / `rl` / `ph` / `ptm` — one u64 BE cell per
12//!   [`WatermarkV1`] field. `chi` (checkpoint) is absent when the committer has not observed
13//!   a checkpoint yet.
14//!
15//! The framework runs a single writer per pipeline role, so regression prevention on the
16//! committer watermark is a read-then-decide pattern rather than a BigTable CAS. Reader and
17//! pruner watermarks are unconditional writes.
18

60impl BigTableConnection<'_> {
61    /// Returns a mutable reference to the underlying BigTable client.
62    pub fn client(&mut self) -> &mut BigTableClient {
63        &mut self.client
64    }
65
66    /// Read a watermark for read-side methods. Enforces the "hide if `checkpoint < reader_lo`
67    /// or `checkpoint == None`" rule and returns the unwrapped checkpoint for callers that
68    /// need it.
69    async fn get_watermark_for_read(
70        &mut self,
71        pipeline: &str,
72    ) -> Result<Option<(WatermarkV1, u64)>> {
73        let row = self.client.get_pipeline_watermark_rows(pipeline).await?;
74        let Some(watermark) = decode_v1(&row)? else {
75            return Ok(None);
76        };
77        let Some(checkpoint_hi_inclusive) = watermark
78            .checkpoint_hi_inclusive
79            .filter(|&cp| cp >= watermark.reader_lo)
80        else {
81            return Ok(None);
82        };
83        Ok(Some((watermark, checkpoint_hi_inclusive)))
84    }
85}

104#[async_trait]
105impl Connection for BigTableConnection<'_> {
106    async fn init_watermark(
107        &mut self,
108        pipeline_task: &str,
109        checkpoint_hi_inclusive: Option<u64>,
110    ) -> Result<Option<InitWatermark>> {
111        // This initial read is to determine if we need to migrate from v0 to v1.
112        let row = self
113            .client
114            .get_pipeline_watermark_rows(pipeline_task)
115            .await?;
116        let existing_v1 = decode_v1(&row)?;
117        let existing_v0 = decode_v0(&row)?;
118
119        // Case 1: row already in the v1 format → return its values, no write.
120        if let Some(wm) = existing_v1 {
121            return Ok(Some(InitWatermark {
122                checkpoint_hi_inclusive: wm.checkpoint_hi_inclusive,
123                reader_lo: Some(wm.reader_lo),
124            }));
125        }
126
127        let initial = if let Some(v0) = existing_v0 {
128            // Case 2: v0-only row → bootstrap a v1 watermark from the v0 committer fields.
129            let reader_lo = v0.checkpoint_hi_inclusive + 1;
130            WatermarkV1 {
131                epoch_hi_inclusive: v0.epoch_hi_inclusive,
132                checkpoint_hi_inclusive: Some(v0.checkpoint_hi_inclusive),
133                tx_hi: v0.tx_hi,
134                timestamp_ms_hi_inclusive: v0.timestamp_ms_hi_inclusive,
135                reader_lo,
136                pruner_hi: reader_lo,
137                pruner_timestamp_ms: 0,
138            }
139        } else {
140            // Case 3: nothing exists → write a fresh row from the framework's input.
141            let reader_lo = checkpoint_hi_inclusive.map_or(0, |cp| cp + 1);
142            WatermarkV1 {
143                epoch_hi_inclusive: 0,
144                checkpoint_hi_inclusive,
145                tx_hi: 0,
146                timestamp_ms_hi_inclusive: 0,
147                reader_lo,
148                pruner_hi: reader_lo,
149                pruner_timestamp_ms: 0,
150            }
151        };
152
153        let write_happened = self
154            .client
155            .create_pipeline_watermark_if_absent(pipeline_task, &initial)
156            .await?;
157
158        let (checkpoint_hi_inclusive, reader_lo) = if write_happened {
159            (initial.checkpoint_hi_inclusive, initial.reader_lo)
160        } else {
161            let row = self
162                .client
163                .get_pipeline_watermark_rows(pipeline_task)
164                .await?;
165            let Some(wm) = decode_v1(&row)? else {
166                bail!(
167                    "watermark for pipeline {} missing after creation",
168                    pipeline_task
169                );
170            };
171            (wm.checkpoint_hi_inclusive, wm.reader_lo)
172        };
173
174        Ok(Some(InitWatermark {
175            checkpoint_hi_inclusive,
176            reader_lo: Some(reader_lo),
177        }))
178    }
179
180    async fn accepts_chain_id(&mut self, pipeline_task: &str, chain_id: [u8; 32]) -> Result<bool> {
181        self.client.accepts_chain_id(pipeline_task, chain_id).await
182    }
183
184    async fn committer_watermark(
185        &mut self,
186        pipeline_task: &str,
187    ) -> Result<Option<CommitterWatermark>> {
188        Ok(self.get_watermark_for_read(pipeline_task).await?.map(
189            |(wm, checkpoint_hi_inclusive)| CommitterWatermark {
190                epoch_hi_inclusive: wm.epoch_hi_inclusive,
191                checkpoint_hi_inclusive,
192                tx_hi: wm.tx_hi,
193                timestamp_ms_hi_inclusive: wm.timestamp_ms_hi_inclusive,
194            },
195        ))
196    }
197
198    async fn set_committer_watermark(
199        &mut self,
200        pipeline_task: &str,
201        watermark: CommitterWatermark,
202    ) -> Result<bool> {
203        let v0 = WatermarkV0 {
204            epoch_hi_inclusive: watermark.epoch_hi_inclusive,
205            checkpoint_hi_inclusive: watermark.checkpoint_hi_inclusive,
206            tx_hi: watermark.tx_hi,
207            timestamp_ms_hi_inclusive: watermark.timestamp_ms_hi_inclusive,
208        };
209        let cells = vec![
210            (col::EPOCH_HI, u64_be(watermark.epoch_hi_inclusive)),
211            (
212                col::CHECKPOINT_HI,
213                u64_be(watermark.checkpoint_hi_inclusive),
214            ),
215            (col::TX_HI, u64_be(watermark.tx_hi)),
216            (
217                col::TIMESTAMP_MS_HI,
218                u64_be(watermark.timestamp_ms_hi_inclusive),
219            ),
220            (col::WATERMARK_V0, Bytes::from(bcs::to_bytes(&v0)?)),
221        ];
222        self.client
223            .cas_write_pipeline_watermark_cells(
224                pipeline_task,
225                col::CHECKPOINT_HI,
226                watermark.checkpoint_hi_inclusive,
227                cells,
228            )
229            .await
230    }
231}

233#[async_trait]
234impl ConcurrentConnection for BigTableConnection<'_> {
235    async fn reader_watermark(&mut self, pipeline: &str) -> Result<Option<ReaderWatermark>> {
236        Ok(self
237            .get_watermark_for_read(pipeline)
238            .await?
239            .map(|(wm, checkpoint_hi_inclusive)| ReaderWatermark {
240                checkpoint_hi_inclusive,
241                reader_lo: wm.reader_lo,
242            }))
243    }
244
245    async fn pruner_watermark(
246        &mut self,
247        pipeline: &'static str,
248        delay: Duration,
249    ) -> Result<Option<PrunerWatermark>> {
250        let Some((watermark, _)) = self.get_watermark_for_read(pipeline).await? else {
251            return Ok(None);
252        };
253        // Compute max(0, (pruner_timestamp + delay) - now). Use u128 to avoid overflow when
254        // summing the two operands, and saturating_sub so we never underflow when the wait
255        // period has already elapsed. saturating_sub is safe because callers treat anything
256        // < 1 the same.
257        let pruner_ready_ms = (watermark.pruner_timestamp_ms as u128) + delay.as_millis();
258        let now_ms = SystemTime::now().duration_since(UNIX_EPOCH)?.as_millis();
259        let wait_for_ms = i64::try_from(pruner_ready_ms.saturating_sub(now_ms))?;
260        Ok(Some(PrunerWatermark {
261            wait_for_ms,
262            reader_lo: watermark.reader_lo,
263            pruner_hi: watermark.pruner_hi,
264        }))
265    }
266
267    async fn set_reader_watermark(
268        &mut self,
269        pipeline: &'static str,
270        reader_lo: u64,
271    ) -> Result<bool> {
272        let now_ms = SystemTime::now().duration_since(UNIX_EPOCH)?.as_millis() as u64;
273        let cells = vec![
274            (col::READER_LO, u64_be(reader_lo)),
275            (col::PRUNER_TIMESTAMP_MS, u64_be(now_ms)),
276        ];
277        self.client
278            .cas_write_pipeline_watermark_cells(pipeline, col::READER_LO, reader_lo, cells)
279            .await
280    }
281
282    async fn set_pruner_watermark(
283        &mut self,
284        pipeline: &'static str,
285        pruner_hi: u64,
286    ) -> Result<bool> {
287        let cells = vec![(col::PRUNER_HI, u64_be(pruner_hi))];
288        self.client
289            .cas_write_pipeline_watermark_cells(pipeline, col::PRUNER_HI, pruner_hi, cells)
290            .await
291    }
292}
293
294fn u64_be(v: u64) -> Bytes {
295    Bytes::copy_from_slice(&v.to_be_bytes())
296}

299mod tests {
300    use super::*;

302    use crate::testing::BigTableEmulator;
303    use crate::testing::INSTANCE_ID;
304    use crate::testing::create_tables;
305    use crate::testing::require_bigtable_emulator;
306
307    const PIPELINE: &str = "pipeline";
308    const EPOCH_HI: u64 = 7;
309    const CHECKPOINT_HI: u64 = 200;
310    const TX_HI: u64 = 42;
311    const TIMESTAMP_MS_HI: u64 = 99;
312
313    /// Spawn a BigTable emulator and return a connected store.
314    async fn store_conn() -> (BigTableEmulator, BigTableStore) {
315        require_bigtable_emulator();
316        let emulator = tokio::task::spawn_blocking(BigTableEmulator::start)
317            .await
318            .unwrap()
319            .unwrap();
320        create_tables(emulator.host(), INSTANCE_ID).await.unwrap();
321        let client = BigTableClient::new_local(emulator.host().to_string(), INSTANCE_ID.into())
322            .await
323            .unwrap();
324        (emulator, BigTableStore::new(client))
325    }
326
327    #[tokio::test]
328    async fn test_accepts_chain_id_first_call_writes_and_accepts() {
329        let (_emulator, store) = store_conn().await;
330        let mut conn = store.connect().await.unwrap();
331
332        let chain_id = [1u8; 32];
333        assert!(conn.accepts_chain_id(PIPELINE, chain_id).await.unwrap());
334    }

338        let (_emulator, store) = store_conn().await;
339        let mut conn = store.connect().await.unwrap();
340
341        let chain_id = [1u8; 32];
342        assert!(conn.accepts_chain_id(PIPELINE, chain_id).await.unwrap());
343        assert!(conn.accepts_chain_id(PIPELINE, chain_id).await.unwrap());
344    }

346    #[tokio::test]
347    async fn test_accepts_chain_id_mismatching_rejects() {
348        let (_emulator, store) = store_conn().await;
349        let mut conn = store.connect().await.unwrap();
350
351        let chain_id_a = [1u8; 32];
352        let chain_id_b = [2u8; 32];
353        assert!(conn.accepts_chain_id(PIPELINE, chain_id_a).await.unwrap());
354        assert!(!conn.accepts_chain_id(PIPELINE, chain_id_b).await.unwrap());
355    }
356
357    #[tokio::test]
358    async fn test_init_watermark_returns_existing_on_conflict() {
359        let (_emulator, store) = store_conn().await;
360        let mut conn = store.connect().await.unwrap();
361
362        conn.init_watermark(PIPELINE, None).await.unwrap();
363        let watermark = CommitterWatermark {
364            epoch_hi_inclusive: EPOCH_HI,
365            checkpoint_hi_inclusive: CHECKPOINT_HI,
366            tx_hi: TX_HI,
367            timestamp_ms_hi_inclusive: TIMESTAMP_MS_HI,
368        };
369        conn.set_committer_watermark(PIPELINE, watermark)
370            .await
371            .unwrap();
372
373        // init must surface the existing committer watermark regardless of the input.
374        let init = conn
375            .init_watermark(PIPELINE, Some(0))
376            .await
377            .unwrap()
378            .unwrap();
379        assert_eq!(init.checkpoint_hi_inclusive, Some(CHECKPOINT_HI));
380        // init_watermark now bootstraps a `reader_lo` when it surfaces an existing new-schema
381        // row — set_committer_watermark leaves the `reader_lo` that init originally wrote
382        // (0 for init(None) + set_committer, which is what happened here).
383        assert_eq!(init.reader_lo, Some(0));
384    }
